function copyCode(el) {
  const text = el.textContent.replace('Copiado!', '').trim();
  navigator.clipboard.writeText(text).then(() => {
    el.classList.add('copied');
    setTimeout(() => el.classList.remove('copied'), 1500);
  });
}
