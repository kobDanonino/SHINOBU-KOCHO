<script>
  function copyText(element) {
    const text = element.textContent;
    navigator.clipboard.writeText(text).then(() => {
      element.classList.add('copied');
      setTimeout(() => {
        element.classList.remove('copied');
      }, 1500);
    }).catch(() => alert('Error al copiar'));
  }
